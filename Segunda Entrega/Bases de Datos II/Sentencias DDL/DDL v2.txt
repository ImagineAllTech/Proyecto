create table Juez(
IDJ int(3),
primary key(IDJ)
);

create table Competencia(
ID int(4) auto_increment,
Tipo set('Individual', 'Equipos'),
Categor√≠a set('12/13', '14/15', '16/17', 'Mayores'),
CatSexo set('Masculino', 'Femenino', 'Otro'),

create table Grupo(
ID int(4),
NumeroGrupo int(3),
Color set('Aka', 'Ao'),
foreign key(ID) references Competencia(ID)
);

create table Escuela(
NombreEscuela varchar(30),
primary key (NombreEscuela)
);

create table Juzga(
IDJ int(3),
CI int(8),
NumeroKata int(3),
Puntaje decimal(3,1),
Ronda int(1),
Foreign key(IDJ) references Juez(IDJ),
Foreign key(CI) references Competidor(CI),
Foreign key(NumeroKata) references Kata(NumeroKata)
);

create table Asiste(
IDJ int(3),
ID int(4),
Foreign key(IDJ) references Juez(IDJ),
Foreign key(ID) references Competencia(ID)
);

create table Forma(
ID int(4),
CI int(8),
Foreign key(ID) references Grupo(ID),
Foreign key(CI) references Competidor(CI)
);

create table Elige(
CI int(8),
NumeroKata int(3),
Foreign key(CI) references Competidor(CI),
Foreign key(NumeroKata) references Kata(NumeroKata)
);

create table Integra(
NombreEscuela varchar(30),
CI int(8),
Dojo varchar(30),
Foreign key(NombreEscuela) references Escuela(NombreEscuela),
Foreign key(CI) references Competidor(CI)
);
